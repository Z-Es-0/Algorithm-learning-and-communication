class DisjointSet:
    def __init__(self, n):
        self.parent = [i for i in range(n + 5)]  # 多开点数组

    def find(self, x):

        root = x
        # 先找到根节点
        while self.parent[root] != root:
            root = self.parent[root]

        # 路径压缩，将路径上的所有节点直接指向根节点
        while self.parent[x] != root:
            next_node = self.parent[x]
            self.parent[x] = root
            x = next_node

        return root

    def union(self, x, y):  # 合并函数
        rootX = self.find(x)
        rootY = self.find(y)
        if rootX != rootY:
            self.parent[rootX] = rootY
def kruskal(g,n):
    ans=0
    g.sort(key=lambda x:x[2])
    q=DisjointSet(n)
    for i in g:
        if q.find(i[0])!=q.find(i[1]):
            ans+=i[2]
            q.union(i[0],i[1])
        else:
            pass
    return ans



def main():
    g=[]
    n,m=map(int,input().split())
    for i in range(m):
        a,b,c=map(int,input().split())
        g.append((a,b,c))
    print(kruskal(g,n))
if __name__ == '__main__':
    main()

import heapq
class DisjointSet:
    def __init__(self, n):
        self.parent = [i for i in range(n + 5)]  # 多开点数组

    def find(self, x):

        root = x
        # 先找到根节点
        while self.parent[root] != root:
            root = self.parent[root]

        # 路径压缩，将路径上的所有节点直接指向根节点
        while self.parent[x] != root:
            next_node = self.parent[x]
            self.parent[x] = root
            x = next_node

        return root

    def union(self, x, y):  # 合并函数
        rootX = self.find(x)
        rootY = self.find(y)
        if rootX != rootY:
            self.parent[rootX] = rootY

def prim(x,r,n):
    op=DisjointSet(n)
    ans=0
    h=[]
    for i in r[x]:
        heapq.heappush(h,i)
    while h:
        x=heapq.heappop(h)
        if op.find(x[1]) !=op.find(x[2]):
            op.union(x[1], x[2])
            ans+=x[0]
            for i in r[x[2]]:
                heapq.heappush(h, i)
    return ans







def main():
    n,m=map(int,input().split())
    r={i:[] for i in range(1,n+1)}

    for i in range(1,m+1):
        a,b,c=map(int,input().split())
        r[a].append((c,a,b))
        r[b].append((c,b,a))
    print(prim(1,r,n))

if __name__ == '__main__':
    main()
